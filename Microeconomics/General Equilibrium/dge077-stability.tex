\section*{Stability}
\begin{itemize}
  \item Tâtonnement price dynamics is a nonlinear differential equation
\end{itemize}

$$
\frac{d \hat{p}}{d t}=\hat{z}(\hat{p}) \text { on } \mathbb{R}_{++}^{L-1}
$$

\begin{itemize}
  \item A solution is a function $\hat{p}:[0, \infty) \rightarrow \mathbb{R}_{++}^{L-1}$ where $\hat{p}(t)$ solves the differential equation.
  \item $\hat{p}^{*}$ is an equilibrium point if $\hat{z}\left(\hat{p}^{*}\right)=0$.\\
$\forall t, \hat{p}(t)=\hat{p}^{*}, \frac{d \hat{p}(t)}{d t}=\hat{z}\left(\hat{p}^{*}\right)=0$.
  \item Do solutions converge to the WE price?
\end{itemize}

\section*{Stability definitions}
Definition 1 (Locally stable) An equilibrium point $\hat{p}^{*}$ is locally stable (LS) if $\forall \epsilon>0, \exists \delta>0:\left[\left\|\hat{p}_{0}-\hat{p}^{*}\right\|<\delta\right]$ implies that any solution $\hat{p}(t), \hat{p}(0) =\hat{p}_{0}$ satisfies $\left\|\hat{p}(t)-\hat{p}^{*}\right\|<\epsilon$

If $\hat{p}^{*}$ is not locally stable, it is unstable.\\
Definition 2 (Locally Asymptotically stable) An equilibrium point $\hat{p}^{*}$ is locally asymptotically stable (LAS) if it is locally stable and every locally stable solution $\hat{p}(t), \hat{p}(0)=\hat{p}_{0}$ satisfies $\lim _{t \rightarrow \infty} \hat{p}(t)=\hat{p}^{*}$

\begin{itemize}
  \item Linear approximation: $\frac{d \hat{p}}{d t}=\left.D \hat{z}\right|_{\hat{p}^{*}}\left(\hat{p}-\hat{p}^{*}\right)$ suffices to identify LAS equilibria.
\end{itemize}

\section*{Stability results}
Theorem 1 (Hirsch and Smale) An equilibrium point $\hat{p}^{*}$ is locally asymptotically stable if and only if all eigenvalues of the Jacobian $D_{\hat{p}} \hat{z}$ evaluated at $\hat{p}^{*}$ have a negative real part.

The proof is in Hirsch and Smale (1974), chapter 9.

\begin{itemize}
  \item If one eigenvalue has a positive real part, $\hat{p}^{*}$ is not LAS.
\end{itemize}
